450        public void addXYSeries(XYMultipleSeriesDataset dataset, String[] titles, List<double[]> xValues,List<double[]> yValues, int scale) {
451                int length = titles.length;
452                for (int i = 0; i < length; i++) {
453                        XYSeries series = new XYSeries(titles[i], scale);

455                        double[] yV = yValues.get(i);
456                        int seriesLength = xV.length;

458                                series.add(xV[k], yV[k]);
